movement_anti_monarchy = {
	category = movement_category_ideological
	
	ideology = ideology_radical_movement
	character_ideologies = {
		ideology_liberal_leader
		ideology_libertarian_leader
		ideology_socialist_leader
		ideology_progressive_leader
		ideology_communist_leader
	}
	
	disband_trigger = {
		NOT = { is_a_monarchy = yes }
	}

	creation_trigger = {
		is_a_monarchy = yes
	}
	creation_weight = {
		value = 100
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	
	can_pressure_interest_group = {
		law_stance = {
			law = law_type:law_HOS_monarch
			value < neutral
		}
	}
	
	character_support_trigger = {
		law_stance = {
			law = law_type:law_HOS_monarch
			value < neutral
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				has_ideology = ideology:ideology_progressive_leader
			}
			multiply = {
				value = 5
			}
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_libertarian_leader
			}
			multiply = {
				value = 5
			}
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_liberal_leader
			}
			multiply = {
				value = 2
			}
		}
	}
	
	pop_support_trigger = {
		NOR = {
			is_pop_type = aristocrats
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_high_literacy
		movement_support_capitalists
		movement_support_aristocrats
		movement_support_officers
		movement_support_academics
		movement_support_bureaucrats
		movement_support_engineers
		movement_support_clerks
		movement_support_soldiers
	}
	
	pop_support_weight = {	
		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}

		if = {
			limit = {
				strata = upper
			}
			if = {
				limit = {
					is_pop_type = capitalists
				}
				add = {
					value = 15
					desc = "POP_CAPITALISTS"
				}
			}
			if = {
				limit = {
					is_pop_type = aristocrats
				}
				if = {
					limit = {
						owner ?= {
							has_journal_entry = je_great_reformer
						}
					}
					add = {
						value = 25
						desc = "POP_ARISTOCRATS"
					}
				}
				else = {
					add = {
						value = 10
						desc = "POP_ARISTOCRATS"
					}
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = officers
				}
				add = {
					value = 15
					desc = "POP_OFFICERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 25
					desc = "POP_ACADEMICS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = bureaucrats
				}
				add = {
					value = 15
					desc = "POP_BUREAUCRATS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = engineers
				}
				add = {
					value = 25
					desc = "POP_BUREAUCRATS"
				}
			}
			else = {
				add = {
					value = 10
					desc = "MIDDLE_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 9
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 6
					desc = "POP_SOLDIERS"
				}
			}
			else = {
				add = {
					value = 3
					desc = "LOWER_NO_ICON"
				}
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}
	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = 1		
		}
	}
	
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 1000
		}		
	}		
}

movement_anti_war = {

	ideology = ideology_anti_war_movement
	character_ideologies = {
		ideology_liberal_leader
		ideology_populist_leader
		ideology_progressive_leader
	}
	
	disband_trigger = {
		is_at_war = no
	}

	creation_trigger = {
		is_at_war = yes
	}
	creation_weight = {
		value = 100
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	
	can_pressure_interest_group = {
		NOR = { 
			is_interest_group_type = ig_armed_forces
			is_interest_group_type = ig_intelligentsia
		}
		leader ?= {
			OR = {
				has_ideology = ideology:ideology_progressive_leader
				has_ideology = ideology:ideology_populist_leader
			}
		}
	}
	
	character_support_trigger = {
		NOR = { 
			is_interest_group_type = ig_armed_forces
			is_interest_group_type = ig_intelligentsia
		}
		OR = {
			has_ideology = ideology:ideology_populist_leader
			has_ideology = ideology:ideology_progressive_leader
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				has_ideology = ideology:ideology_populist_leader
			}
			multiply = {
				value = 5
			}
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_progressive_leader
			}
			multiply = {
				value = 2
			}
		}
	}
	
	pop_support_trigger = {
		NOR = {
			is_pop_type = aristocrats
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_literacy
		movement_support_below_expected_sol
	}
	
	pop_support_weight = {
		if = {
			limit = {
				standard_of_living < state.average_expected_sol # It is intentional that a labourer being well-off doesn't decrease support for this.
			}
			add = { # Increases likelihood of support if impoverished.
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}
		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}
	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = 1		
		}
	}
	
	secession = {
		possible = {
			always = no
		}
		
		weight = {
			value = 1000
		}
	}
	
}

movement_collaborationist = {
	category = movement_category_ideological
	
	ideology = ideology_absolutist_movement
	character_ideologies = {
		ideology_traditionalist_leader
	}
	
	disband_trigger = {
		OR = { 
			is_sufficiently_independent = yes
			NOT = { exists = scope:overlord_scope }
		}
	}
	
	creation_trigger = {
		NOT = { is_sufficiently_independent = yes }
	}
	creation_weight = {
		value = 100
	}	
	on_created = {
		add_movement_enthusiasm_modifier = yes
		owner = { 
			overlord = { 
				save_scope_as = overlord_scope
			}
		}
	}
	
	can_pressure_interest_group = {
		leader ?= { 
			culture = {
				is_primary_culture_of = scope:overlord_scope
			}
		}
	}
	
	character_support_trigger = {
		culture = {
			is_primary_culture_of = scope:overlord_scope
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				has_ideology = ideology:ideology_autocratic
			}
			multiply = {
				value = 5
			}
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_power_hungry
			}
			multiply = {
				value = 3
			}
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_elitist
			}
			multiply = {
				value = 1
			}
		}
	}

	pop_support_trigger = {
		culture = {
			is_primary_culture_of = scope:overlord_scope
		}
		NOT = { is_pop_type = slaves }
	}

	pop_support_factors = {
		movement_support_colonial_rule
		movement_support_high_acceptance
		movement_support_bureaucrats
		movement_support_academics
		movement_support_officers
		movement_support_clerks
	}
	
	pop_support_weight = {
		value = 1
		
		if = {
			limit = {
				culture = {
					is_primary_culture_of = scope:overlord_scope
				}
			}
			add = {
				desc = "colonial_rule"
				value = 15
				if = {
					limit = {
						is_pop_type = bureaucrats
					}
					multiply = 1.5
				}
			}
		}

		if = {
			limit = {
				is_pop_type = bureaucrats
			}
			add = {
				desc = "POP_BUREAUCRATS"
				value = 9
			}
		}
		else_if = {
			limit = {
				is_pop_type = academics
			}
			add = {
				desc = "POP_ACADEMICS"
				value = 6
			}
		}
		else_if = {
			limit = {
				is_pop_type = capitalists
			}
			add = {
				desc = "POP_CAPITALISTS"
				value = 3
			}
		}
		else_if = {
			limit = {
				is_pop_type = officers
			}
			add = {
				desc = "POP_OFFICERS"
				value = 3
			}
		}
		else_if = {
			limit = {
				is_pop_type = clerks
			}
			add = {
				desc = "POP_CLERKS"
				value = 3
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = {
			desc = "POP_ACCEPTANCE"
			value = pop_acceptance
			divide = 75
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}	
	
		weight = {
			value = 100
		}
	
		state_weight = {
			value = 1
			if = {
				limit = { 
					OR = {
					    any_scope_building = {
							is_building_group = bg_bureaucracy
							level >= 10
					    }		
					}
				}
				multiply = 10
			}
			else_if = {
				limit = { 
					OR = {
					    any_scope_building = {
							is_building_group = bg_bureaucracy
							level >= 5
					    }		
					}
				}
				multiply = 5
			}
			else = {
				multiply = 0
			}
		}
	}
	
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 1000
		}		
		
		state_weight = {
			value = 1
			if = {
				limit = { 
					OR = {
					    any_scope_building = {
							is_building_group = bg_bureaucracy
							level >= 10
					    }		
					}
				}
				multiply = 10
			}

							is_building_group = bg_bureaucracy
							level >= 5
					    }		
					}
				}
				multiply = 5
			}
			else = {
				multiply = 0
			}
		}		
	}	
	
	additional_radicalism_factors = {
		owner = {
			if = {
				limit = { has_institution = institution_schools }
				
				add = {
					value = 3
					subtract = "institution_investment_level(institution_schools)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_schools"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_schools"
				}		
			}				
		}
	}

movement_corporatist = {
	category = movement_category_ideological
	
	ideology = ideology_libertarian_leader
	character_ideologies = {
		ideology_libertarian_leader
		ideology_conservative_leader
		ideology_populist_leader
	}
	
	creation_trigger = {
		
	}	
	creation_weight = {
		value = 100
	}	
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	
	character_support_trigger = {
		OR = { 
			
			#Politician ideologies
			interest_group ?= { 
				OR = { 
					has_ideology = ideology:ideology_right_economy
					has_ideology = ideology:ideology_centre_economy
				}
			}
			
			#Agitator ideologies
			has_ideology = ideology:ideology_agitator_pro_war_rights
			has_ideology = ideology:ideology_agitator_nationalist
			has_ideology = ideology:ideology_agitator_anti_human_rights
			
			has_ideology = ideology:ideology_autocratic
			has_ideology = ideology:ideology_power_hungry
			has_ideology = ideology:ideology_elitist
			
		}
	}
	character_support_weight = {
		if = {
			limit = {
				has_ideology = ideology:ideology_autocratic
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				interest_group ?= { has_ideology = ideology:ideology_militarist }
			}
			multiply = {
				value = 2
			}
		}
	}
	
	pop_support_trigger = {
		NOT = {
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_literacy
		movement_support_middle_below_expected_sol
		movement_support_radicalization
		movement_support_unemployed
		movement_support_capitalists
		movement_support_aristocrats
		movement_support_shopkeepers
		movement_support_officers
		movement_support_farmers
		movement_support_clerks
		movement_support_soldiers
	}
	
	pop_support_weight = {	
		add = {
			value = 3
			desc = "POP_BASE_SUPPORT"
		}

		if = {
			limit = {
				strata = upper
			}
			if = {
				limit = {
					is_pop_type = capitalists
				}
				add = {
					value = 9
					desc = "POP_CAPITALISTS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = aristocrats
				}
				add = {
					value = 3
					desc = "POP_ARISTOCRATS" # Fascism is too populist
				}
			}
			else = {
				add = {
					value = 6
					desc = "UPPER_NO_ICON"
				}
			}

			if = {
				limit = {
					owner ?= {
						any_political_movement = {
							OR = {
								is_political_movement_type = movement_socialist
								is_political_movement_type = movement_libertarian
								is_political_movement_type = movement_communist
							}
							political_movement_support >= 0.15 # (chuckles) im in danger
						}
					}
				}
				multiply = {
					value = 1.5
					desc = "STRONG_SOCIALIST_MOVEMENT"
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = shopkeepers
				}
				add = {
					value = 15
					desc = "POP_SHOPKEEPERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = officers
				}
				add = {
					value = 12
					desc = "POP_OFFICERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = farmers
				}
				add = {
					value = 12
					desc = "POP_FARMERS"
				}
			}
			else = {
				add = {
					value = 9
					desc = "MIDDLE_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 9
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 6
					desc = "POP_SOLDIERS"
				}
			}
			else = {
				add = {
					value = 3
					desc = "LOWER_NO_ICON"
				}
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
				strata = middle
			}
			add = { # Increases likelihood of support if impoverished
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}
		else_if = {
			limit = {
				standard_of_living > state.average_expected_sol
				strata = middle
			}
			add = { # Decreases likelihood of support if well-off
				desc = "IS_ABOVE_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				min = -5
			}
		}

		if = {
			limit = {
				is_employed = no
			}
			multiply = { 
				value = 1.5
				desc = "UNEMPLOYED"
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = {
			value = 0.012
			multiply = pop_acceptance
			desc = "ACCEPTANCE"
		}
	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = 1		
		}
	}
	
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 1000
		}		
	}	
	
	additional_radicalism_factors = {
		owner = {
			if = {
				limit = {
					owner.number_of_claims > 0
				}
				
				add = {
					value = 5 # Add a base value for having claims at all
					add = owner.number_of_claims
					multiply = 0.025
					max = 0.3
					desc = "NUMBER_OF_CLAIMS"
				}				
			}
		}
	}	
}
