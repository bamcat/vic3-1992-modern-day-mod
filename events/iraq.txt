namespace = iraq_events

iraq_events.1 = {
    type = country_event
    placement = ROOT
    title = iraq_events.10.t
    desc = iraq_events.10.d
    event_image = { video = "fighterjets" }
	duration = 3
    trigger = {
        c:IRQ ?= this
        game_date >= 1993.1.1
        NOT = { has_variable = iraq_fly_happened }
    }

    immediate = { }

    option = {
        name = iraq_events.10.a
		default_option = yes
        set_variable = unscom_aircraft_rejected
		set_variable = iraq_fly_happened 
        change_relations = {
            country = c:USA
            value = -25
        }
    }

    option = {
        name = iraq_events.10.b
		set_variable = iraq_fly_happened 
        change_relations = {
            country = c:USA
            value = 10
        }
    }
}
