haiti_events.1 = {
    type = country_event
    placement = ROOT
    title = haiti_events.1.t
    desc = haiti_events.1.d
    event_image = { video = "helicopter" }
	duration = 2
    trigger = {
		exists = c:HAI
        OR = {
            c:USA ?= this
            c:HAI ?= this
        }
        game_date >= 1994.9.1
        NOT = { has_variable = haiti_intervention_1994_happened }
    }

    immediate = { set_variable = haiti_intervention_1994_happened }

    option = {
        name = haiti_events.1.a
        trigger = {
            c:USA ?= this
        }
		c:HAI = { 
        country_event = { id = haiti_events.2 days = 3 } # Notify Haiti
		}
    }

    option = {
        name = haiti_events.1.b
        trigger = {
            c:HAI ?= this
        }
    }
}
