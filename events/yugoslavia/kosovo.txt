# Collapse of Yugoslavia

namespace = kosovo_events

# Kosovo war
kosovo_events.1 = {
    type = country_event
    placement = ROOT
    title = kosovo_events.1.t
    desc = kosovo_events.1.d
    event_image = { video = "yugoslav_war" }
	duration 2
    trigger = {
        c:SRB ?= this
		game_date >= 1998.2.1
        NOT = { global_var:kosovo_conflict_active = 1 }
    }

    immediate = {
        set_variable = kosovo_conflict_active
        set_global_variable = {
            name = kosovo_war_escalation
            value = 1
        }
    }

    option = {
        name = kosovo_events.1.a
		c:ALB = { 
		set_global_variable = { name = kosovo_conflict_active value = 1 }
		create_diplomatic_play = {
			name = kosovo_war
			type = dp_annex_war
			target_country = c:SRB
			
			add_war_goal = { holder = c:ALB type = conquer_state target_state = scope:s:STATE_SOUTHERN_SERBIA primary_demand = yes target_country = c:SRB }
			add_war_goal = {
				holder = c:SRB
				type = annex_country
				target_country = c:ALB
			}
		}
        ai_chance = { base = 100 }
    }
 }
}

kosovo_events.2 = {
    type = country_event
    placement = ROOT
    title = kosovo_events.2.t
    desc = kosovo_events.2.d
    event_image = { video = "fighterjets" }
	duration = 2
    trigger = {
        NOT = { has_variable = nato_bombing_started }
		global_var:kosovo_conflict_active = 1
		game_date >= 1998.8.1
        OR = {
            c:USA ?= this
            c:GBR ?= this
            c:FRA ?= this
        }
    }

	immediate = { set_variable = nato_bombing_started }
    option = {
        name = kosovo_events.2.a
		default_option = yes
		c:SRB = { trigger_event = { id = kosovo_events.3 popup = yes } }
        every_country = {
            limit = { has_variable = kosovo_conflict_active }
            add_radicals = { value = 0.05 }
        }
        c:SRB = { set_variable = nato_bombing_started }
    }
	
    option = {
        name = kosovo_events.2.b
        add_loyalists = { value = 0.02 }
    }
}
