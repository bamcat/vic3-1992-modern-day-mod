# War Crimes in Bosnia

namespace = yugoslav_war_crimes

# --- War Crime 1: Siege of Sarajevo ---
yugoslav_war_crimes.1 = {
    type = country_event
    title = yugoslav_war_crimes.1.t
    desc = yugoslav_war_crimes.1.d
    event_image = { video = "yugoslav_war" }
	
    trigger = {
		NOT = { has_variable = sarajevo_happened }
        OR = {
			c:SRB ?= this
			c:SRP ?= this
		}
		c:BOS = {
			enemy_occupation > 0.20
		}
        game_date >= 1992.5.1
    }

    option = {
        name = yugoslav_war_crimes.1.a
		set_variable = sarajevo_happened
        add_modifier = {
            name = international_condemnation
            months = 12
        }
        add_radicals = {
            value = 0.05
            pop_type = laborers
        }
        add_relations = { target = c:USA value = -25 }
        add_relations = { target = c:GBR value = -25 }
        add_relations = { target = c:FRA value = -25 }
		every_country = { limit = { NOT = { OR = {
			c:SRB ?= this
			c:SRP ?= this
		} } }
        trigger_event = { id = yugoslav_war_crimes.3 days = 3 }
		}
		c:BOS = { trigger_event = { id = yugoslav_war_crimes.5 days = 3 } }
    }
}
